import React, { useState, useEffect, useRef, useCallback } from 'react'; import { initializeApp } from 'firebase/app'; import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from 'firebase/auth'; import { getFirestore, collection, doc, setDoc, updateDoc, onSnapshot, getDoc, deleteDoc } from 'firebase/firestore'; import { Car, Zap, Trophy, AlertTriangle, Users, Play, Copy, LogOut } from 'lucide-react'; // --- CONFIGURATION FIREBASE --- const firebaseConfig = JSON.parse(__firebase_config); const app = initializeApp(firebaseConfig); const auth = getAuth(app); const db = getFirestore(app); const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id'; // --- CONSTANTES DU JEU --- const LANE_COUNT = 3; const CAR_WIDTH = 50; const CAR_HEIGHT = 80; const CANVAS_WIDTH = 400; // Largeur fixe pour simplifier la logique responsive const CANVAS_HEIGHT = 600; const OBSTACLE_SPEED_BASE = 5; const SYNC_INTERVAL = 500; // Synchroniser avec la DB toutes les 500ms // --- UTILITAIRES --- const generateRoomId = () => Math.floor(1000 + Math.random() * 9000).toString(); // --- AUTHENTIFICATION --- useEffect(() => { const initAuth = async () => { try { if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) { await signInWithCustomToken(auth, __initial_auth_token); } else { await signInAnonymously(auth); } } catch (err) { console.error("Auth error", err); setError("Erreur de connexion au serveur."); } }; initAuth(); return onAuthStateChanged(auth, (u) => { if (u) { setUser(u); const unsubscribe = onSnapshot(roomRef, (snapshot) => { if (snapshot.exists()) { const data = snapshot.data(); setPlayers(data.players || {}); // Démarrage automatique si 2 joueurs sont prêts if (view === 'ROOM' && data.status === 'RACING') { startGameLocal(); } // Vérification de fin de partie if (view === 'GAME' && data.players) { checkEndGameCondition(data.players); } } else { // La salle a été supprimée if (view !== 'LOBBY') handleLeaveRoom(); } }, (err) => { console.error("Erreur sync salle:", err); setError("Perte de connexion à la salle."); }); return () => unsubscribe(); }, [roomId, user, view]); const createRoom = async () => { if (!user) return; const newRoomId = generateRoomId(); const roomRef = doc(db, 'artifacts', appId, 'public', 'data', `races_${newRoomId}`); try { await setDoc(roomRef, { host: user.uid, status: 'WAITING', createdAt: Date.now(), players: { [user.uid]: { name: 'Joueur 1', id: user.uid, distance: 0, isAlive: true, status: 'WAITING' // WAITING, READY, RACING, CRASHED } } }); setRoomId(newRoomId); setView('ROOM'); } catch (e) { setError("Impossible de créer la salle."); } }; const joinRoom = async (inputRoomId) => { if (!user || !inputRoomId) return; const roomRef = doc(db, 'artifacts', appId, 'public', 'data', `races_${inputRoomId}`); try { const snap = await getDoc(roomRef); if (snap.exists()) { const data = snap.data(); if (Object.keys(data.players).length >= 2) { setError("La salle est pleine."); return; } await updateDoc(roomRef, { [`players.${user.uid}`]: { name: 'Joueur 2', id: user.uid, distance: 0, isAlive: true, status: 'WAITING' } }); setRoomId(inputRoomId); setView('ROOM'); } else { setError("Salle introuvable."); } } catch (e) { setError("Erreur en rejoignant la salle."); } }; const setReady = async () => { if (!roomId || !user) return; const roomRef = doc(db, 'artifacts', appId, 'public', 'data', `races_${roomId}`); try { // Si nous sommes l'hôte et que tout le monde est prêt (ou juste moi pour test solo), on lance const isHost = players[user.uid]?.name === 'Joueur 1'; const allReady = Object.values(players).every(p => p.status === 'READY' || p.id === user.uid); // Simple check logic let newStatus = 'READY'; // Logique simple: si je clique prêt, je passe en READY. // Si les deux sont READY, le backend (ou l'hôte) passe en RACING. // Ici, on fait simple: l'hôte lance quand il clique si l'autre est là. await updateDoc(roomRef, { [`players.${user.uid}.status`]: 'READY' }); // Vérification pour lancer la course (côté client hôte) if (isHost && Object.keys(players).length > 0) { // On lance la course après un court délai pour laisser l'update passer setTimeout(async () => { await updateDoc(roomRef, { status: 'RACING' }); }, 500); } } catch (e) { console.error(e); } }; const handleLeaveRoom = async () => { setView('LOBBY'); setPlayers({}); setRoomId(''); setGameResult(null); // Nettoyage optionnel de la DB non inclus pour simplifier (laisser les salles expirer ou supprimer si hôte) }; // --- MOTEUR DE JEU (CANVAS) --- const startGameLocal = () => { setView('GAME'); setGameResult(null); gameStateRef.current = { active: true, playerX: 1, obstacles: [], distance: 0, speed: OBSTACLE_SPEED_BASE, lastSync: 0, isAlive: true, particles: [] }; requestRef.current = requestAnimationFrame(gameLoop); }; const checkEndGameCondition = (currentPlayers) => { const myPlayer = currentPlayers[user.uid]; const opponents = Object.values(currentPlayers).filter(p => p.id !== user.uid); const opponent = opponents[0]; if (!myPlayer) return; // Si je suis mort if (!gameStateRef.current.isAlive && gameStateRef.current.active) { endGameLocal('CRASHED'); return; } // Si l'adversaire est mort et que je suis vivant if (opponent && !opponent.isAlive && opponent.status === 'CRASHED' && gameStateRef.current.isAlive) { // J'ai gagné par survie endGameLocal('WIN_SURVIVAL'); } }; const endGameLocal = async (reason) => { gameStateRef.current.active = false; cancelAnimationFrame(requestRef.current); // Mettre à jour mon statut final if (roomId && user) { const roomRef = doc(db, 'artifacts', appId, 'public', 'data', `races_${roomId}`); await updateDoc(roomRef, { [`players.${user.uid}.isAlive`]: reason !== 'CRASHED', [`players.${user.uid}.status`]: 'CRASHED', // Marqueur de fin [`players.${user.uid}.finalDistance`]: gameStateRef.current.distance }); } if (reason === 'CRASHED') { setGameResult('LOSE'); } else if (reason === 'WIN_SURVIVAL') { setGameResult('WIN'); } setView('GAME_OVER'); }; // Gestion des contrôles useEffect(() => { const handleKeyDown = (e) => { if (view !== 'GAME') return; }, [view]); const movePlayer = (dir) => { if (!gameStateRef.current.active) return; gameStateRef.current.playerX = Math.max(0, Math.min(2, gameStateRef.current.playerX + dir)); }; const updatePhysics = () => { const state = gameStateRef.current; // Augmenter la vitesse progressivement state.speed += 0.005; state.distance += state.speed; // Générer obstacles if (Math.random() < 0.02 + (state.speed * 0.001)) { const lanes = [0, 1, 2]; const lane = lanes[Math.floor(Math.random() * lanes.length)]; // Éviter de spawner sur une ligne déjà occupée tout en haut const tooClose = state.obstacles.some(o => o.y < 150 && o.lane === lane); if (!tooClose) { state.obstacles.push({ lane, y: -100, type: Math.random() > 0.8 ? 'ROCK' : 'BARRIER' }); } } // Bouger obstacles state.obstacles.forEach(o => { o.y += state.speed * 1.5; // Ils viennent vers nous }); // Nettoyer obstacles passés state.obstacles = state.obstacles.filter(o => o.y < CANVAS_HEIGHT + 100); const collisionY = (o.y + 40 > playerY) && (o.y < playerY + CAR_HEIGHT); return sameLane && collisionY; }); if (crash) { state.isAlive = false; // Créer des particules for(let i=0; i<10; i++) { state.particles.push({ x: (playerLane * (CANVAS_WIDTH/3)) + 50, y: playerY + 40, vx: (Math.random() - 0.5) * 10, vy: (Math.random() - 0.5) * 10, life: 30, color: 'orange' }); } } // Sync DB (Throttle) const now = Date.now(); if (now - state.lastSync > SYNC_INTERVAL && roomId && user) { const roomRef = doc(db, 'artifacts', appId, 'public', 'data', `races_${roomId}`); // Fire and forget pour la perf updateDoc(roomRef, { [`players.${user.uid}.distance`]: Math.floor(state.distance) }).catch(e => {}); // Ignorer erreurs silencieuses pdt le jeu state.lastSync = now; } }; const draw = (ctx) => { const state = gameStateRef.current; // Fond ctx.fillStyle = '#111827'; // Dark gray/blue ctx.fillRect(0, 0, CANVAS_WIDTH, CANVAS_HEIGHT); // Effet de vitesse (lignes sur les côtés) ctx.strokeStyle = '#374151'; ctx.lineWidth = 2; const offset = (state.distance * 10) % 50; // Lignes de séparation des voies ctx.setLineDash([30, 30]); ctx.lineDashOffset = -offset; // Voie 1 ctx.beginPath(); ctx.moveTo(CANVAS_WIDTH / 3, 0); ctx.lineTo(CANVAS_WIDTH / 3, CANVAS_HEIGHT); ctx.stroke(); // Voie 2 ctx.beginPath(); ctx.moveTo((CANVAS_WIDTH / 3) * 2, 0); ctx.lineTo((CANVAS_WIDTH / 3) * 2, CANVAS_HEIGHT); ctx.stroke(); ctx.setLineDash([]); // Reset // Dessiner Joueur if (state.isAlive) { const px = (state.playerX * (CANVAS_WIDTH / 3)) + (CANVAS_WIDTH / 6) - (CAR_WIDTH / 2); const py = CANVAS_HEIGHT - CAR_HEIGHT - 20; // Ombre (Glow) ctx.shadowBlur = 15; ctx.shadowColor = '#00f2ff'; ctx.fillStyle = '#00f2ff'; // Cyan néon // Forme voiture simple ctx.beginPath(); ctx.roundRect(px, py, CAR_WIDTH, CAR_HEIGHT, 5); ctx.fill(); // Détails voiture ctx.fillStyle = '#000'; ctx.fillRect(px + 5, py + 10, CAR_WIDTH - 10, 15); // Pare-brise ctx.shadowBlur = 0; // Reset } // Dessiner Obstacles state.obstacles.forEach(o => { const ox = (o.lane * (CANVAS_WIDTH / 3)) + (CANVAS_WIDTH / 6) - (CAR_WIDTH / 2); ctx.shadowBlur = 10; ctx.shadowColor = '#ff0055'; ctx.fillStyle = '#ff0055'; // Rouge néon if (o.type === 'ROCK') { ctx.beginPath(); ctx.arc(ox + CAR_WIDTH/2, o.y + CAR_WIDTH/2, CAR_WIDTH/2, 0, Math.PI * 2); ctx.fill(); } else { ctx.fillRect(ox, o.y, CAR_WIDTH, 30); } ctx.shadowBlur = 0; }); // Particules (Explosion) state.particles.forEach((p, idx) => { p.x += p.vx; p.y += p.vy; p.life--; ctx.fillStyle = p.color; ctx.globalAlpha = p.life / 30; ctx.fillRect(p.x, p.y, 5, 5); ctx.globalAlpha = 1; if (p.life <= 0) state.particles.splice(idx, 1); }); }; const gameLoop = () => { const ctx = canvasRef.current?.getContext('2d'); if (!ctx) return; if (gameStateRef.current.isAlive) { updatePhysics(); } else if (gameStateRef.current.particles.length === 0 && gameStateRef.current.active) { // Fin de l'animation de mort, on arrête le jeu si ce n'est pas déjà fait par la logique // Mais ici on attend que l'état React prenne le relais via checkEndGameCondition } draw(ctx); // Mettre à jour l'UI React (Score/Vitesse) de temps en temps pour éviter trop de renders // Astuce: On ne met pas à jour le state React à chaque frame (60fps), mais on pourrait. // Pour simplifier ici, on update juste si active. if (gameStateRef.current.active && gameStateRef.current.distance % 10 < 1) { setScore(Math.floor(gameStateRef.current.distance)); setSpeed(Math.floor(gameStateRef.current.speed * 20)); // Vitesse fictive } if (gameStateRef.current.active || gameStateRef.current.particles.length > 0) { requestRef.current = requestAnimationFrame(gameLoop); } }; // --- RENDU UI --- // Calculer la progression de l'adversaire const getOpponentProgress = () => { if (!players || !user) return 0; const opponent = Object.values(players).find(p => p.id !== user.uid); if (!opponent) return 0; // On normalise arbitrairement pour la barre: Disons que 100% de la barre = +200m d'avance ou retard // C'est relatif. On affiche plutôt deux barres absolues si possible, ou juste le delta. return opponent.distance; }; return (
